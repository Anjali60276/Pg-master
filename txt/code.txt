
import 'package:flutter/material.dart';
import 'package:go_router/go_router.dart';

class ShellLayout extends StatelessWidget {
  final Widget child; // child variable , Stores the screen that should appear above the bottom navigation.


  const ShellLayout({
    super.key,
    required this.child, // requires child widgets when creating the shellLayout.
  });

  @override
  Widget build(BuildContext context /* need for navigation*/   ) { //Flutter calls this whenever it needs to draw the UI.
    final location = GoRouterState.of(context).uri.path; //Gets the current url path example: /tasks /schedule /profile

    int index = 0; //default selection tasks

    if (location.startsWith('/schedule')) {
      index = 1; //if the current page is /schedule then index is 1
    } else if (location.startsWith('/profile')) {
      index = 2; //if the current page is /profile then index is 2
    } // its like task = 0,scheduled=1,profile=2

    return Scaffold(            //provide the page structure
      backgroundColor: const Color(0xFFF5F5F5), //sets page background
      body: child, //display whatever page passed out into ShellLayout

    
    );
  }
}

Card or Container for the outer box

Column for the title and subtitle
Chip or a styled Container for the status badge
